export interface AbilityProps {
    name: string;
    abbr: string;
    base: number;
    adjust: number;
    modifier: number;
    changeAbility: (ability: string, delta: number) => void;
}
export const Ability = ({ name, abbr, base, adjust, modifier, changeAbility }: AbilityProps) => {
    const increase = () => {
        changeAbility(abbr, 1)
    }
    const decrease = () => {
        changeAbility(abbr, -1)
    }
    return(
        <div className="ability">
            <div title={name}>{abbr}</div>
            <div className="grid grid-cols-4 gap-2 grid-rows-2 grid-flow-col text-center">
                <span>Score</span>
                <span>{base + adjust + modifier}</span>
                <span>base</span>
                <span>{base}</span>
                <span>adjust</span>
                <span>{adjust}</span>
                <span>modifier</span>
                <span>{modifier}</span>
                <button className="btn-primary" onClick={increase}>+</button>
                <button className="btn-secondary" onClick={decrease}>-</button>
            </div>
        </div>
    )
}
